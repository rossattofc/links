            function showScreen(screenId) {
                const footer = document.getElementById('main-footer');
                
                document.querySelectorAll('.screen').forEach(screen => {
                    screen.classList.remove('active');
                });
                
                const activeScreen = document.getElementById(screenId);
                if (activeScreen) {
                    activeScreen.classList.add('active');
                }

                if (screenId === 'welcome-screen') {
                    footer.style.display = 'flex';
                    if (document.getElementById('quiz-results')) {
                        document.getElementById('quiz-results').style.display = 'none';
                    }
                    if(document.getElementById('finish-module-btn')) {
                        document.getElementById('finish-module-btn').style.display = 'none';
                    }
                    if(document.getElementById('quiz-nav-buttons')) {
                        document.getElementById('quiz-nav-buttons').style.display = 'flex';
                    }
                    document.querySelectorAll('.options label').forEach(label => {
                        label.classList.remove('selected', 'correct', 'incorrect');
                    });
                    document.querySelectorAll('input[type="radio"]').forEach(radio => {
                        radio.checked = false;
                    });
                } else {
                    footer.style.display = 'none';
                }
                
                window.scrollTo(0, 0);

            function buildTimeline(containerId, data) {
                const container = document.getElementById(containerId);
                if (!container) return;
                container.innerHTML = '';
                
                data.forEach((item, index) => {
                    const positionClass = index % 2 === 0 ? 'left' : 'right';
                    const timelineItem = document.createElement('div');
                    timelineItem.className = `timeline-item ${positionClass}`;
                    
                    timelineItem.innerHTML = `
                        <div class="timeline-header">
                            <h3><span class="timeline-icon">${item.icon}</span>${item.title}</h3>
                        </div>
                        <div class="timeline-content">
                            <div class="timeline-content-inner">
                                ${item.content}
                            </div>
                        </div>
                    `;
                    container.appendChild(timelineItem);
                });
            }

            function toggleTimelineItem(header) {
                const timelineItem = header.closest('.timeline-item');
                const content = timelineItem.querySelector('.timeline-content');
                const isAlreadyActive = timelineItem.classList.contains('active');

                // Fecha todos os outros itens abertos
                document.querySelectorAll('.timeline-item.active').forEach(activeItem => {
                    if (activeItem !== timelineItem) {
                        activeItem.classList.remove('active');
                        activeItem.querySelector('.timeline-content').style.maxHeight = null;
                    }
                });

                if (!isAlreadyActive) {
                    timelineItem.classList.add('active');
                    content.style.maxHeight = content.scrollHeight + "px";
                } else {
                    timelineItem.classList.remove('active');
                    content.style.maxHeight = null;
                }
            }

            function showQuizResults() {
                let correctCount = 0;
                const questions = document.querySelectorAll('.question-container');
                
                questions.forEach(question => {
                    const optionsDiv = question.querySelector('.options');
                    const correctAnswer = optionsDiv.getAttribute('data-answer');
                    const selectedRadio = question.querySelector('input[type="radio"]:checked');
                    
                    optionsDiv.querySelectorAll('label').forEach(l => {
                        l.classList.remove('correct', 'incorrect');
                    });

                    if (selectedRadio) {
                        const selectedLabel = selectedRadio.parentElement;
                        if (selectedRadio.value === correctAnswer) {
                            correctCount++;
                            selectedLabel.classList.add('correct');
                        } else {
                            selectedLabel.classList.add('incorrect');
                            optionsDiv.querySelector(`input[value="${correctAnswer}"]`).parentElement.classList.add('correct');
                        }
                    }
                });

                const totalQuestions = questions.length;
                const percentage = (correctCount / totalQuestions) * 100;
                const resultsEl = document.getElementById('quiz-results');
                resultsEl.textContent = `Voc√™ acertou ${correctCount} de ${totalQuestions} perguntas (${percentage.toFixed(0)}%).`;
                resultsEl.style.display = 'block';
                
                document.getElementById('quiz-nav-buttons').style.display = 'none';
                document.getElementById('finish-module-btn').style.display = 'inline-block';

                // ENVIA A NOTA PARA O SCORM
                if(typeof scorm !== 'undefined') {
                    scorm.complete(percentage.toFixed(0));
                }
            }
